A simple function

Let's calculate the circumference and area of a circle \[\begin{aligned} C & = 2 \pi r \\ A & = \pi r^2 \end{aligned}\] 

						#!/usr/bin/python
						import numpy as np

						def circle_info(r):
							# Get circumference and area of a circle
							C = 2 * np.pi * r
							A = np.pi * r ** 2
							return C, A

						print(circle_info(2))
